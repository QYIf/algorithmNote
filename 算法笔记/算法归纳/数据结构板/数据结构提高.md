##  1.并查集

###  1) 普通并查集
```cpp
//一般的并查集
const int N = 20010;
//分别是父结点数组，节点
int p[N], n;

//初始化父结点
void init() {
    for(int i =1; i <= n; i++) p[i] = i;
}

//找到当前点的父结点
int find(int x) {
    if(x == p[x]) return x;
    return p[x] = find(p[x]);
}

//连通两个集合
void merge(int x, int y) {
    if(find(x) != find(y))
    	p[find(x)] = find(y);
}
```

```cpp
//维护子集大小的并查集
const int N = 200010;
int p[N], size[N], n;

//初始化父结点和集合的大小（只含有自己所以是1）
void init() {
    for(int i = 1; i <= n; i++) {

        size[i] = 1;

    return p[x] = find(p[x]);
}

    	p[find(x)] = find(y);
    }
}
```

### 3) 维护边权的并查集
```cpp
//维护边权的并查集
const int N = 200010;
int p[N], d[N], n;

void init() {
    for(int i = 1; i <= n; i++) {
        p[i] = i;
        d[i] = 0;
    }
}

int find(int x) {
    if(x == p[x]) return x;
    int tmp = find(p[x]);
    d[x] += d[p[x]];
    return p[x] = tmp;
}

void merge(int x, int y) {
    if(find(x) != find(y)) {
    	p[find(x)] = find(y);
    	d[find(x)] = d[y] - d[x] + n; //n与具体题目有关
    }
}
```

### 1) 单点修改

```cpp
//此时数组维护的是当前的数组（即数组内保存的是其本身的值）
int a[N], b[N], t[N];
int n, m;

//lowbit操作
int lb(int x) {return x & -x;}

//在idx点上加上num的值
int add(int idx, int num) {
	for(; idx <= n; idx += lb(idx)) t[idx] += num;
} 

//查询某个点的前缀和
int ask(int x) {
	int ans = 0;
	for(; x; x -= lb(x)) ans += t[x];
	return ans;
}

//查询y点的值
ask(y) - ask(y - 1);
//查询l到r区间的和
```

## 3.线段树

### 1) 区间查询，单点修改

```cpp
using namespace std;
const int N = 500010;
int m, p, arr[N];
char op[10];

//线段树数组（记得开四倍空间）
struct Node {
	int l, r;
	int sum;
    //还有可能包含其他信息
} tr[N * 4];

//更新合并两个点
void pushup(Node &cur, Node &l, Node &r) {
	cur.sum = l.sum + r.sum;
    //还有可能还有其他的信息需要合并
}

//用于单点更新
void pushup(int u) {
	pushup(tr[u], tr[u << 1], tr[u << 1 | 1]);
}

//创建树
void build(int u, int l, int r) {
    //当前点为根结点时，赋上对应的值
	if (l == r) {
		tr[u] = {l, r, arr[l]};
	} else {
        //只赋予左右区间的范围，具体的值由pushup来传递
		tr[u].l = l, tr[u].r = r;
        //获取中点
		int mid = l + r >> 1;
        //分开两个区间进行更新
		build(u << 1, l, mid), build(u << 1 | 1, mid + 1, r);
        //拿子结点更新父结点
		pushup(u);
	}
}

//区间查询
Node query(int u, int l, int r) {
    //当查到的区间在查询范围内，直接拿当前值
	if (tr[u].l >= l && tr[u].r <= r) return tr[u];
	else {
		int mid = tr[u].l + tr[u].r >> 1;
        //整个区间在左边
		if (r <= mid) {
			return query(u << 1, l, r);
            //整个区间在右边
		} else if (l > mid) {
			return query(u << 1 | 1, l, r);
            //两边都有
		} else {
            Node tmp;
            //获取左边区间
            Node left = query(u << 1, l, r);
            //获取右边区间
            Node right = query(u << 1 | 1, l, r);
            //合并两区间
            pushup(tmp, left, right);
            return  tmp;
		}
	}
}

//单点修改
void modify(int u, int x, int val) {
    //当找到了这个点时，对其进行修改
	if (tr[u].l == x && tr[u].r == x) tr[u].sum = val;
	else {
		int mid = tr[u].l + tr[u].r >> 1;
        //如果这个点在左边区间就往左找
		if (x <= mid) modify(u << 1, x, val);
        //如果这个点在右边区间就往右找
		else modify(u << 1 | 1, x, val);
        //拿子结点更新父结点(与建树的原理差不多)
		pushup(u);
	}
}
```

### 2) 区间查询，区间修改

```cpp
#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#define ll long long
using namespace std;
const ll N = 500010;

ll n, m, a, b, c, w[N];
char op;

struct Node{
    ll l, r;
    ll sum, tag;
}tr[4 * N];

//每次修改（建立）子节点后更新
void pushup(ll u) {
    tr[u].sum = tr[u << 1].sum + tr[u << 1 | 1].sum;
}

//在往下访问子节点时，看看当前结点的懒标记是否有东西，若存在东西就将懒标记下移
//注意懒标记内的东西在下传到某个点时，当前点所带的值要跟着更新，并留下懒标记
void pushdown(ll u) {
    Node &cur = tr[u];
    Node &l = tr[u << 1];
    Node &r = tr[u << 1 | 1];
    if(cur.tag) {
        l.tag += cur.tag, l.sum += (l.r - l.l + 1) * cur.tag;
        r.tag += cur.tag, r.sum += (r.r - r.l + 1) * cur.tag;
        cur.tag = 0;
    }
}

//建立线段树
void build(ll u, ll l, ll  r) {
    //当递归到叶子节点时，对其赋值（懒标记也要）
    if(l == r) {
        tr[u] = {l, r, w[l], 0};
    } else {
        tr[u] = {l, r};
        ll mid = (l + r) >> 1;
        //对两个子节点进行建树
        build(u << 1, l, mid), build(u << 1 | 1, mid + 1, r);
        //每次建好子节点后，对子节点进行更新
        pushup(u);
    }
}

//查询某个区间的值
ll ask(ll u, ll l, ll r) {
    //当前位置在查询范围之内，可以直接返回
    if(tr[u].l >= l && tr[u].r <= r) {
        return tr[u].sum;
    } else {
        //在往下查询到子节点时，必须要将当前结点的懒标记信息往下传递
        pushdown(u);
        ll mid = (tr[u].l + tr[u].r) >> 1;
        ll sum = 0;
        //获取有需要的值并返回
        if(r > mid) sum += ask(u << 1 | 1, l, r);
        if(l <= mid) sum += ask(u << 1, l, r);
        return sum;
    }
}

//修改某个区间的值
void modify(ll u, ll l, ll r, ll v) {
    //当前位置在查询范围之内，可以直接把修改的信息交给懒标记，并更新当前点的信息
    if(tr[u].l >= l && tr[u].r <= r) {
        tr[u].sum += (tr[u].r - tr[u].l + 1) * v;
        tr[u].tag += v;
    } else {
        //在往下修改到子节点时，必须要将当前结点的懒标记信息往下传递
        pushdown(u);
        ll mid = (tr[u].l + tr[u].r) >> 1;
        //对于区间的分布进行讨论
        if(r > mid) modify(u << 1 | 1, l, r, v);
        if(l <= mid) modify(u << 1, l, r, v);
        //处理完子节点后，对当前结点进行更新
        pushup(u);
    }
}



int main() {
    scanf("%lld%lld", &n, &m);
    for(int i = 1; i <= n; i++) scanf("%lld", &w[i]);
    build(1, 1, n);
    while(m--) {
        scanf(" %c", &op);
        if(op == 'Q') {
            scanf("%lld%lld", &a, &b);
            printf("%lld\n", ask(1, a, b));
        } else {
            scanf("%lld%lld%lld", &a, &b, &c);
            modify(1, a, b, c);
        }
    }
}
```

### 3) 扫描线

```cpp
#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<vector>
using namespace std;
const int N = 100010;

int n, idx, k;
double x1, x2, y1, y2;
//用于离散化
vector<double> alls;

//用于存取区间
struct Segment{
    //y1, y2是对应的x的区间的两个端点
    double x, y1, y2;
    int k;
    //这里是用于排序的
    bool operator< (const Segment &t)const {
        return x < t.x;
    }
}seg[2 * N];

struct Node {
    int l, r;
    //用于记录
    int cnt;
    double len;
}tr[8 * N];

//用于寻找离散化后对应的坐标
int find(double a) {
    return lower_bound(alls.begin(), alls.end(), a) - alls.begin();
}

//更新父结点
void pushup(int u) {
    //如果当前点是被标记了已覆盖的话，我们直接让当前区间的长度等于其左右端点的距离
    //这里注意我们的线段树其实维护的东西是线段而不是点，此时我们要获取长度所以要找两个点相减
    //这里我们从l线段到r线段的距离就等于alls[tr[u].r + 1] - alls[tr[u].l]，因为这段距离的点数比线段数多一
    if(tr[u].cnt) tr[u].len = alls[tr[u].r + 1] - alls[tr[u].l];
    else tr[u].len = tr[u << 1].len + tr[u << 1 | 1].len;
}

//建立线段树
void build(int u, int l, int r) {
    if(l == r) {
        tr[u] = {l, r, 0, 0};
    } else {
        tr[u] = {l, r};
        int mid = (l + r) >> 1;
        build(u << 1, l, mid), build(u << 1 | 1, mid + 1, r);
        //此处非常特殊不用pushup那是因为一开始什么东西都没有存进来
    }
}

//修改线段树的某个区间
void modify(int u, int l, int r, int v) {
    //如果当前区间是在要求区间之内，则直接对这个区间加标记
    if(tr[u].l >= l && tr[u].r <= r) {
        tr[u].cnt += v;
    } else {
        //分类讨论是否需要修改
        int mid = (tr[u].l + tr[u].r) >> 1;
        if(l <= mid) modify(u << 1,l, r, v);
        if(r > mid) modify(u << 1 | 1,l, r, v);
    }
    //子结点完成修改后，要对当前结点进行更新
    pushup(u);
}

int main() {
    while(~scanf("%d", &n) && n) {
        alls.clear();
        idx = 0;
        for(int i = 0; i < n; i++) {
            scanf("%lf%lf%lf%lf", &x1, &y1, &x2, &y2);
            seg[idx++] = {x1, y1, y2, 1};
            seg[idx++] = {x2, y1, y2, -1};
            alls.push_back(y1), alls.push_back(y2);
        }
        //排序去重（离散化的步骤）
        sort(alls.begin(), alls.end());
        alls.erase(unique(alls.begin(), alls.end()), alls.end());
        //建树（这里总共n - 1条线段）
        build(1, 0, alls.size() - 2);
        //对区间进行排序
        sort(seg, seg + n * 2);
        double ans = 0;
        //先把第一个边读进去
        modify(1, find(seg[0].y1), find(seg[0].y2) - 1, seg[0].k);
        for(int i = 1; i < 2 * n; i++) {
            //每次读下一条边然后进行计算
            ans += tr[1].len * (seg[i].x - seg[i - 1].x);
            //把这个边的信息进行处理
            modify(1, find(seg[i].y1), find(seg[i].y2) - 1, seg[i].k);
        }
        printf("Test case #%d\n", ++k);
        printf("Total explored area: %.2lf\n\n", ans);
    }
}
```



## 4. ST表

```cpp
#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
#define ll long long
using namespace std;
const ll N = 200010;
const ll M = 18;
ll n, m;
ll a[N];
//第一维表示起点
//第二维表示区域长度为(2 ^ j)
ll st[N][M];

//先打表初始化所有的st表
void getst() {
    for(ll j = 0; j < M; j++) {
        for(ll i = 1; i + (1 << j) - 1 <= n; i++) {
            //如果j为0代表i自己本身，初始化为a[i];
            if(!j) st[i][j] = a[i];
            //由于更小的区间已经被更新过了，所以我们找到更小的区间（即将当前区间分成两半）然后求出两段区间的最值
            else st[i][j] = max(st[i][j - 1], st[i + (1 << j - 1)][j - 1]);
        }
    }
}

//取一段2的k次方的长度使得两段长度可以覆盖到当前区间的所有点
//然后取两段区间的最大值
ll query(ll l, ll r) {
    ll k = log(r - l + 1) / log(2);
    return max(st[l][k], st[r - (1 << k) + 1][k]);
}

int main() {
    scanf("%lld", &n);
    for(ll i = 1; i <= n; i++) scanf("%lld", &a[i]);
    getst();
    scanf("%lld", &m);
    while(m--) {
        ll a, b;
        scanf("%lld%lld", &a, &b);
        printf("%lld\n", query(a, b));
    }
}
```
