# 数据结构进阶

可持续化数据结构主要用于记录不同版本下的数据

## 可持续化Trie树

```cpp
#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int N = 600010, M = N * 25;

int tr[M][2], root[N], idx;
int n, m, tmp, s[N], mxi[M];
int l, r, x;
char op;

//插入操作
void insert(int i, int k, int p, int q) {
    //当前位为负数时说明插入完成
    if(k < 0) {
        mxi[q] = i;
        return;
    }
    //获取第k位是1还是0
    int v = s[i] >> k & 1;
    //其他分支直接从上个版本中拷贝过来
    if(p) tr[q][v ^ 1] = tr[p][v ^ 1];
    //该分支新建点
    tr[q][v] = ++idx;
    //继续往下插入
    insert(i, k - 1, tr[p][v], tr[q][v]);
    //往上更新最大点
    mxi[q] = max(mxi[tr[q][0]], mxi[tr[q][1]]);
}

//查询操作
int ask(int rt, int num, int l) {
    //从某个头节点开始
    int cur = rt;
    //从最高位开始往下查找
    for(int i = 23; i >= 0; i--) {
        int v = num >> i & 1;
        if(mxi[tr[cur][v ^ 1]] >= l) cur = tr[cur][v ^ 1];
        else cur = tr[cur][v];
    }
    return num ^ s[mxi[cur]];
}

int main() {
    scanf("%d%d", &n, &m);
    //初始化
    mxi[0] = -1;
    root[0] = ++idx;
    //插入哨兵
    insert(0, 23, 0, root[0]);
    for(int i = 1; i <= n; i++) {
        scanf("%d", &x);
        //前缀异或和
        s[i] = s[i - 1] ^ x;
        //插入
        root[i] = ++idx;
        insert(i, 23, root[i - 1], root[i]);
    }
    while(m--) { 
        scanf(" %c", &op);
        if(op == 'A') {
            scanf("%d", &x);
            n++;
            s[n] = s[n - 1] ^ x;
            root[n] = ++idx;
            insert(n, 23, root[n - 1], root[n]);
        } else {
            scanf("%d%d%d", &l, &r, &x);
            printf("%d\n", ask(root[r - 1], s[n] ^ x, l - 1));
        }
    }
    return 0;
}
```

## 可持续化线段树 (主席树)

```cpp
#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<vector>
using namespace std;
const int N = 100010;

int n, m;
int a[N], root[N], idx;
int l, r, k;
vector<int> alls;

//树节点
struct Node {
    int l, r;
    int cnt;
}tr[21 * N];

//离散化找点
int find(int x) {
    return lower_bound(alls.begin(), alls.end(), x) - alls.begin();
}

//建树
int build(int l, int r) {
    int p = ++idx;
    if(l == r) return p;
    int mid = l + r >> 1;
    //分别创建左右子树
    tr[p].l = build(l, mid), tr[p].r = build(mid + 1, r);
    return p;
}

//插入
int insert(int p, int l, int r, int x) {
    int q = ++idx;
    //先把之前的版本复制过来
    tr[q] = tr[p];
    //当递归到单个点时，直接修改某个点的信息
    if(l == r) {
        tr[q].cnt++;
        return q;
    }
    int mid = l + r >> 1;
    //更新左子树
    if(x <= mid) tr[q].l = insert(tr[p].l, l, mid, x);
    //更新右子树
    else tr[q].r = insert(tr[p].r, mid + 1, r, x);
    //pushup
    tr[q].cnt = tr[tr[q].l].cnt + tr[tr[q].r].cnt;
    return q;
}


//查询
int ask(int q, int p, int l, int r, int k) {
    //查询到单点时返回节点
    if(l == r) return l;
    int mid = l + r >> 1;
    //找到当前区间中左子树的数量
    int cnt = tr[tr[q].l].cnt - tr[tr[p].l].cnt;
    //找左子树
    if(k <= cnt) return ask(tr[q].l, tr[p].l, l, mid, k);
    //找右子树
    else return ask(tr[q].r, tr[p].r, mid + 1, r, k - cnt);
}

int main() {
    scanf("%d%d", &n, &m);
    for(int i = 1; i <= n; i++) scanf("%d", &a[i]), alls.push_back(a[i]);
    sort(alls.begin(), alls.end());
    alls.erase(unique(alls.begin(), alls.end()), alls.end());
    root[0] = build(0, alls.size() - 1);
    for(int i = 1; i <= n; i++) root[i] = insert(root[i - 1], 0, alls.size() - 1, find(a[i]));
    while(m--) {
        scanf("%d%d%d", &l, &r, &k);
        printf("%d\n", alls[ask(root[r], root[l - 1], 0, alls.size() - 1, k)]);
    }
}
```

## 平衡树

### (1) Treap

```cpp
#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int N = 200010, INF = 1e8;

int n, root;
int op, v, idx;

struct Node {
    //左右子树
    int l, r;
    //key是要存的值，val是用于排序的值
    int key, val;
    //cnt表示当前值的数量，size表示当前子树的大小
    int cnt, size;
} tr[N];

//往上更新
void pushup(int u) {
    tr[u].size = tr[tr[u].l].size + tr[tr[u].r].size + tr[u].cnt;
}

//这里穿引用是为了方便后面调用时保证当前树的相对结构

//左旋
void zig(int& p) {
    int q  = tr[p].l;
    tr[p].l = tr[q].r, tr[q].r = p, p = q;
    pushup(tr[p].r), pushup(p);
}

//右旋
void zag(int& p) {
    int q = tr[p].r;
    tr[p].r = tr[q].l, tr[q].l = p, p = q;
    pushup(tr[p].l), pushup(p);
}

//创建新节点，相当于初始化方法
int nnew(int d) {
    //记录点的值
    tr[++idx].key = d;
    //随即可以时树更加平衡
    tr[idx].val = rand();
    //初始化大小
    tr[idx].cnt = tr[idx].size = 1;
    //返回节点值
    return idx;
}

//建树建立哨兵
void build() {
    nnew(-INF), nnew(INF);
    root = 1, tr[1].r = 2;
    if(tr[1].val < tr[2].val) zag(root);
    pushup(root);
}

//插入操作
void insert(int& u, int key) {
    //如果插入的位置没有点，则把这个点放在当前位置
    if(!u) {
        u = nnew(key);
        //当前位置的值等于要插入的值时直接让当前点的数量加一
    } else if(tr[u].key == key) {
        tr[u].cnt++;
        //小于则往左插入
    } else if(tr[u].key < key) {
        insert(tr[u].r, key);
        if(tr[tr[u].r].val > tr[u].val) zag(u);
        //大于则往右插入
    } else {
        insert(tr[u].l, key);
        if(tr[tr[u].l].val > tr[u].val) zig(u);
    }
    //最后一定要往上更新
    pushup(u);
}

//删除结点
void del(int& u, int key) {
    //如果当前结点为空则不进行任何操作
    if(!u) return;
    //当前值与val相等时开始考虑删除
    else if(tr[u].key == key) {
        //如果数量大于1则数量减少一即可
        if(tr[u].cnt != 1) tr[u].cnt--;
        //如果当前点有子树的情况下，使结点往下放
        else if(tr[u].r || tr[u].l) {
            if(!tr[u].r || tr[tr[u].l].val > tr[tr[u].r].val) {
                zig(u);
                del(tr[u].r, key);
            } else {
                zag(u);
                del(tr[u].l, key);
            }
        }
        //若没有子树，则直接删除该点即可
        else u = 0;
    }
    //往左找删除点
    else if(tr[u].key < key) del(tr[u].r, key);
    //往右找删除点
    else del(tr[u].l, key);
    //往上更新
    pushup(u);
}

//通过值找到排名
int ra(int u, int key) {
    if(!u) return 0;
    if(tr[u].key == key) return tr[tr[u].l].size + 1;
    if(tr[u].key < key) return tr[tr[u].l].size + tr[u].cnt + ra(tr[u].r, key);
    else return ra(tr[u].l, key);
}

//通过排名找值
int val(int u, int rr) {
    if(!u) return -INF;
    if(tr[tr[u].l].size >= rr) return val(tr[u].l, rr);
    if(tr[tr[u].l].size + tr[u].cnt >= rr) return tr[u].key;
    return val(tr[u].r, rr - tr[tr[u].l].size - tr[u].cnt);
}

//找到某个点的前驱
int pre(int u, int v) {
    if(!u) return -INF;
    if(tr[u].key >= v) return pre(tr[u].l, v);
    return max(tr[u].key, pre(tr[u].r, v));
}

//找到某个点的后继
int ne(int u, int v) {
    if(!u) return INF;
    if(tr[u].key <= v) return ne(tr[u].r, v);
    return min(tr[u].key, ne(tr[u].l, v));
}

int main() {
    scanf("%d", &n);
    build();
    while(n--) {
        scanf("%d%d", &op, &v);
        if(op == 1) {
            insert(root, v);
        } else if(op == 2) {
            del(root, v);
        } else if(op == 3) {
            printf("%d\n", ra(root, v) - 1);
        } else if(op == 4) {
            printf("%d\n", val(root, v + 1));
        } else if(op == 5) {
            printf("%d\n", pre(root, v));
        } else {
            printf("%d\n", ne(root, v));
        }
    }
}
```

## 分块

```cpp
#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
#define ll long long
using namespace std;
const int N = 200010;

ll n, m, len;
ll a[N], sum[N], tag[N];
ll l, r, d;
char op;

//获取某个点属于哪个分块
ll get(ll idx) {
    return idx / len;
}

//对某个区间进行操作
void add(ll l, ll r, ll val) {
    ll la = get(l), ra = get(r);
    ll i = l, j = r;
    //如果区间处于一个块中，则直接对这个区间暴力处理
    if(la == ra) {
        for(ll k = l; k <= r; k++) a[k] += val, sum[la] += val;
    //反之，把头尾非整区间进行处理后，对后面的分块进行整块处理
    } else {
        while(la == get(i)) a[i] += val, sum[la] += val, i++;
        while(ra == get(j)) a[j] += val, sum[ra] += val, j--;
        for(ll k = la + 1 ; k <= ra - 1; k++) tag[k] += val, sum[k] += len * val;
    }
}

//区间查询
ll ask(ll l, ll r) {
    ll la = get(l), ra = get(r);
    ll i = l, j = r, ans = 0;
    //如果区间处于一个块中，则直接对这个区间暴力处理
    if(la == ra) {
        for(ll k = l; k <= r; k++) ans += a[k] + tag[la];
    //反之，把头尾非整区间进行累加后，对后面的分块进行整块处理
    } else {
        while(la == get(i)) ans += a[i] + tag[la], i++;
        while(ra == get(j)) ans += a[j] + tag[ra], j--;
        for(ll k = la + 1 ; k <= ra - 1; k++) ans += sum[k];
    }
    return ans;
}

int main() {
    scanf("%lld%lld", &n, &m);
    len = sqrt(n);
    //预处理好每个区间的和
    for(int i = 1; i <= n; i++) scanf("%lld", &a[i]), sum[get(i)] = sum[get(i)] + a[i];
    while(m--) {
        scanf(" %c", &op);
        if(op == 'Q') {
            scanf("%lld%lld", &l, &r);
            printf("%lld\n", ask(l, r));
        } else {
            scanf("%lld%lld%lld", &l, &r, &d);
            add(l, r, d);
        }
    }
}
```

## 莫队 

### 基础莫队

```cpp
#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;
const int N = 1000010;

int n, m, num, len;
int a[N];
int cnt[N], ans[N];
int l, r;

//获取当亲点位于哪个分块之中
int get(int x) {
    return x / len;
}

//查询
struct Query {
    int idx, l, r;
    //查询左端点按分块查询，右端点按单调顺序排
    bool operator < (const Query& w) const {
        int ll = get(l), rr = get(w.l);
        if(ll == rr) return r < w.r;
        return ll < rr;
    }
} q[N];

//对当前点进行加操作
void add(int x, int &num) {
    if(!cnt[x]) num++;
    cnt[x]++;
}

//对当前点进行删除操作
void del(int x, int &num) {
    cnt[x]--;
    if(!cnt[x]) num--;
}

int main() {
    scanf("%d", &n);
    for(int i = 1; i <= n; i++) scanf("%d", &a[i]);
    scanf("%d", &m);
    len = sqrt((double)n * n / m);
    for(int i = 1; i <= m; i++) scanf("%d%d", &l, &r), q[i] = {i, l, r};
    sort(q + 1, q + m + 1);
    //j作为左端点，i作为右端点
    for(int i = 1, j = 2, k = 1; k <= m; k++) {
        int ll = q[k].l, rr = q[k].r, id = q[k].idx;
        //根据要求对两个指针进行前后移动
        while (i < rr) add(a[++i], num);
        while (i > rr) del(a[i--], num);
        while (j < ll) del(a[j++], num);
        while (j > ll) add(a[--j], num);
        ans[id] = num;
    }
    for (int i = 1; i <= m; i++) printf("%d\n", ans[i]);
}
```
