# MKの线段树题单

[【模板】线段树 1](https://www.luogu.com.cn/problem/P3372)

题解：线段树区间修改区间查询模板题

```cpp
#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<set>
#define ll long long
using namespace std;
const int N = 200010;

struct Node {
    ll l, r;
    ll sum, tag;
} tr[4 * N];

ll n, m;
ll op, x, y, k;
ll w[N];

void pushup(ll u) {
    tr[u].sum = tr[u << 1].sum + tr[u << 1 | 1].sum;
}

void pushdown(ll u) {
    Node &cur = tr[u];
    Node &l = tr[u << 1];
    Node &r = tr[u << 1 | 1];
    if(cur.tag) {
        l.tag += cur.tag, l.sum += (l.r - l.l + 1) * cur.tag;
        r.tag += cur.tag, r.sum += (r.r - r.l + 1) * cur.tag;
        cur.tag = 0;
    }
}

void build(ll u, ll l, ll  r) {
    if(l == r) {
        tr[u] = {l, r, w[l], 0};
    } else {
        tr[u] = {l, r};
        ll mid = (l + r) >> 1;
        build(u << 1, l, mid), build(u << 1 | 1, mid + 1, r);
        pushup(u);
    }
}

ll ask(ll u, ll l, ll r) {
    if(tr[u].l >= l && tr[u].r <= r) {
        return tr[u].sum;
    } else {
        pushdown(u);
        ll mid = (tr[u].l + tr[u].r) >> 1;
        ll sum = 0;
        if(r > mid) sum += ask(u << 1 | 1, l, r);
        if(l <= mid) sum += ask(u << 1, l, r);
        return sum;
    }
}

void modify(ll u, ll l, ll r, ll v) {
    if(tr[u].l >= l && tr[u].r <= r) {
        tr[u].sum += (tr[u].r - tr[u].l + 1) * v;
        tr[u].tag += v;
    } else {
        pushdown(u);
        ll mid = (tr[u].l + tr[u].r) >> 1;
        if(r > mid) modify(u << 1 | 1, l, r, v);
        if(l <= mid) modify(u << 1, l, r, v);
        pushup(u);
    }
}

int main() {
    scanf("%lld%lld", &n, &m);
    for(ll i = 1; i <= n; i++) scanf("%lld", &w[i]);
    build(1, 1, n);
    while(m--) {
        scanf("%lld%lld%lld", &op, &x, &y);
        if(op == 1) {
            scanf("%lld", &k);
            modify(1, x, y, k);
        } else {
            printf("%lld\n", ask(1, x, y));
        }
    }
}
```

[统计和](https://www.luogu.com.cn/problem/P2068)

题解：简单的单点修改，区间查询

```cpp
#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#define ll long long
using namespace std;
const int N = 200010;

struct Node {
    ll l, r;
    ll sum;
} tr[4 * N];

ll n, m;
char op;
ll x, y;

void pu(ll u) {
    tr[u].sum = tr[u << 1].sum + tr[u << 1 | 1].sum;
}

void build(ll u, ll l, ll r) {
    tr[u] = {l, r, 0};
    if(l == r) return;
    ll mid = (l + r) >> 1;
    build(u << 1, l, mid), build(u << 1 | 1, mid + 1, r);
    pu(u);
}

void modify(ll u, ll idx, ll val) {
    if(tr[u].l == idx && tr[u].r == idx) {
        tr[u].sum += val;
        return;
    }
    ll mid = (tr[u].l + tr[u].r) >> 1;
    if(mid >= idx) modify(u << 1, idx, val);
    else modify(u << 1 | 1, idx, val);
    pu(u);
}

ll ask(ll u, ll l, ll r) {
    if(tr[u].l >= l && tr[u].r <= r) {
        return tr[u].sum;
    }
    ll mid = (tr[u].l + tr[u].r) >> 1;
    ll sum = 0;
    if(l <= mid) sum += ask(u << 1, l, r);
    if(r > mid) sum += ask(u << 1 | 1, l, r);
    return sum;
}

int main() {
    cin >> n >> m;
    build(1, 1, n);
    while(m--) {
        cin >> op >> x >> y;
        if(op == 'x') {
            modify(1, x, y);
        } else {
            printf("%lld\n", ask(1, x, y));
        }
    }
}
```

[I Hate It](https://www.luogu.com.cn/problem/P1531)

题解：单点修改，区间维护最大值

```cpp
#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int N = 200010;

int n, m;
int w[N];
int x, y;
char op;

struct Node{
    int l, r;
    int mx;
} tr[4 * N];

void pushup(int u) {
    tr[u].mx = max(tr[u << 1].mx, tr[u << 1 | 1].mx);
}

void build(int u, int l, int r) {
    tr[u] = {l, r, 0};
    if(l == r) {
        tr[u].mx = w[l];
        return;
    }
    int mid = l + r >> 1;
    build(u << 1, l, mid), build(u << 1 |  1, mid + 1 , r);
    pushup(u);
}

void modify(int u, int idx, int val) {
    if(tr[u].l == idx && tr[u].r == idx) {
        tr[u].mx = max(tr[u].mx, val);
        return;
    }
    int mid = tr[u].l + tr[u].r >> 1;
    if(idx <= mid) modify(u << 1, idx, val);
    else modify(u << 1 | 1, idx, val);
    pushup(u);
}

int ask(int u, int l, int r) {
    if(tr[u].l >= l && tr[u].r <= r) {
        return tr[u].mx;
    }
    int mid = tr[u].l + tr[u].r >> 1;
    int mx = 0;
    if(l <= mid) mx = max(ask(u << 1, l, r), mx);
    if(r > mid) mx = max(ask(u << 1 | 1, l, r), mx);
    return mx;
}

int main() {
    scanf("%d%d", &n, &m);
    for(int i = 1; i <= n; i++) scanf("%d", &w[i]);
    build(1, 1, n);
    while(m--) {
        scanf(" %c%d%d", &op, &x, &y);
        if(op == 'Q') {
            printf("%d\n", ask(1, x, y));
        } else {
            modify(1, x, y);
        }
    }
}
```
